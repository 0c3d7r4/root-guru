 #include <vector>
 #include <queue>
 #include <climits>
 using namespace std;

 void dijkstra(int graph[][5], int start) {
     vector<int> dist(5, INT_MAX);  // Массив расстояний
     dist[start] = 0;

     priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
     pq.push({0, start});

     while (!pq.empty()) {
         int u = pq.top().second;
         pq.pop();

         for (int v = 0; v < 5; v++) {
             if (graph[u][v] && dist[u] != INT_MAX && dist[u] + graph[u][v] < dist[v]) {
                 dist[v] = dist[u] + graph[u][v];
                 pq.push({dist[v], v});
             }
         }
     }

     for (int i = 0; i < 5; i++) {
         cout << "Distance from " << start << " to " << i << " is " << dist[i] << endl;
     }
 }

 int main() {
     int graph[5][5] = {
         {0, 2, 0, 1, 0},
         {2, 0, 3, 2, 0},
         {0, 3, 0, 0, 4},
         {1, 2, 0, 0, 5},
         {0, 0, 4, 5, 0}
     };

import heapq

 def dijkstra(n, a, start, end, c):
     # Using Dijkstra's to find the minimum maximum path cost
     INF = float('inf')
     dist = [[INF] * n for _ in range(n)]
     dist[start[0]][start[1]] = a[start[0]][start[1]]
     pq = [(a[start[0]][start[1]], start[0], start[1])]  # (cost, x, y)

     directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]  # up, down, left, right

     while pq:
         max_cost, x, y = heapq.heappop(pq)

         if (x, y) == end:
             return max_cost

         for dx, dy in directions:
             nx, ny = x + dx, y + dy
             if 0 <= nx < n and 0 <= ny < n:
                 new_cost = max(max_cost, a[nx][ny])
                 if new_cost < dist[nx][ny]:
                     dist[nx][ny] = new_cost
                     heapq.heappush(pq, (new_cost, nx, ny))

     return dist[end[0]][end[1]]
